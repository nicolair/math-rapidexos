\begin{displaymath}
 (1+x^p)^q \sim u(x)
\end{displaymath}

\begin{displaymath}
 (1+x^p)^q - u(x)
\end{displaymath}

\begin{displaymath}
 (1+x^p)^q - (1+x^q)^p
